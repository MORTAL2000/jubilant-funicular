148  template<typename T, qualifier Q>
149  GLM_FUNC_QUALIFIER tquat<T, Q>::tquat(vec<3, T, Q> const& u, vec<3, T, Q> const& v)
150  {
151  T norm_u_norm_v = sqrt(dot(u, u) * dot(v, v));
152  T real_part = norm_u_norm_v + dot(u, v);
153  vec<3, T, Q> t;
154 
155  if(real_part < static_cast<T>(1.e-6f) * norm_u_norm_v)
156  {
157  // If u and v are exactly opposite, rotate 180 degrees
158  // around an arbitrary orthogonal axis. Axis normalisation
159  // can happen later, when we normalise the quaternion.
160  real_part = static_cast<T>(0);
161  t = abs(u.x) > abs(u.z) ? vec<3, T, Q>(-u.y, u.x, static_cast<T>(0)) : vec<3, T, Q>(static_cast<T>(0), -u.z, u.y);
162  }
163  else
164  {
165  // Otherwise, build quaternion the standard way.
166  t = cross(u, v);
167  }
168 
169  *this = normalize(tquat<T, Q>(real_part, t.x, t.y, t.z));
170  }

635  Result[1][2] = T(2) * (qyz + qwx);
636 
637  Result[2][0] = T(2) * (qxz + qwy);
638  Result[2][1] = T(2) * (qyz - qwx);

649  template<typename T, qualifier Q>
650  GLM_FUNC_QUALIFIER tquat<T, Q> quat_cast(mat<3, 3, T, Q> const& m)
651  {
652  T fourXSquaredMinus1 = m[0][0] - m[1][1] - m[2][2];
653  T fourYSquaredMinus1 = m[1][1] - m[0][0] - m[2][2];
654  T fourZSquaredMinus1 = m[2][2] - m[0][0] - m[1][1];
655  T fourWSquaredMinus1 = m[0][0] + m[1][1] + m[2][2];
656 
657  int biggestIndex = 0;
658  T fourBiggestSquaredMinus1 = fourWSquaredMinus1;
659  if(fourXSquaredMinus1 > fourBiggestSquaredMinus1)
660  {
661  fourBiggestSquaredMinus1 = fourXSquaredMinus1;
662  biggestIndex = 1;
663  }
664  if(fourYSquaredMinus1 > fourBiggestSquaredMinus1)
665  {
666  fourBiggestSquaredMinus1 = fourYSquaredMinus1;
667  biggestIndex = 2;
668  }
669  if(fourZSquaredMinus1 > fourBiggestSquaredMinus1)
670  {
671  fourBiggestSquaredMinus1 = fourZSquaredMinus1;
672  biggestIndex = 3;
673  }
674 
675  T biggestVal = sqrt(fourBiggestSquaredMinus1 + static_cast<T>(1)) * static_cast<T>(0.5);
676  T mult = static_cast<T>(0.25) / biggestVal;
677 
678  switch(biggestIndex)
679  {
680  case 0:
681  return tquat<T, Q>(biggestVal, (m[1][2] - m[2][1]) * mult, (m[2][0] - m[0][2]) * mult, (m[0][1] - m[1][0]) * mult);
682  case 1:
683  return tquat<T, Q>((m[1][2] - m[2][1]) * mult, biggestVal, (m[0][1] + m[1][0]) * mult, (m[2][0] + m[0][2]) * mult);
684  case 2:
685  return tquat<T, Q>((m[2][0] - m[0][2]) * mult, (m[0][1] + m[1][0]) * mult, biggestVal, (m[1][2] + m[2][1]) * mult);
686  case 3:
687  return tquat<T, Q>((m[0][1] - m[1][0]) * mult, (m[2][0] + m[0][2]) * mult, (m[1][2] + m[2][1]) * mult, biggestVal);
688  default: // Silence a -Wswitch-default warning in GCC. Should never actually get here. Assert is just for sanity.
689  assert(false);
690  return tquat<T, Q>(1, 0, 0, 0);
691  }
692  }
