14 /*
15 ** This file was created by Jesse Maurais Wed April 4, 2007
16 ** Contact: www.jessemaurais@gmail.com
17 **
18 **
19 ** This patch to the Devil Tookit binds to the X Windows System Version 11,
20 ** using client-side XImages and server-side Pixmaps, with support for both
21 ** ZPixmaps and the insane XYPixmap format. (Depends on the X server)
22 **
23 ** If the XShm extension to X11 is present at the time of ./configure then
24 ** support for shared memory XImages and Pixmaps is also compiled in. Note
25 ** that "shared" does not mean Devil and X are sharing the same memory space.
26 ** This is not possible, because both libraries make byte-for-byte copies
27 ** of their data. It means that memory is part of an inter-process memory
28 ** segment (see XShm spec).
29 **
30 ** TODO
31 ** 1) Assumed the display depth is 24 bits (the most common) but there should
32 ** be a check, and iXConvertImage should handle this properly. I don't think
33 ** this should be difficult to modify from whats here.
34 ** 2) It would be nice to convert from an XImage back to a Devil image for
35 ** saving changes. Would be seful for an interactive image editor.
36 ** 3) Possibly some additional OpenGL bindings for GLX Pbuffers.
37 **
38 ** FYI
39 ** It was a long night figuring out the under-documented XYPixmap format.
40 */

82 void iXConvertImage( Display * dpy, XImage * img )
83 {
84  int x,y,z;
85  int sX,dX;
86  int sY,dY;
87  int sZ,dZ;
88  int plane;
89 
90 
91  ILimage * tmp;
92 
93  switch ( img->byte_order )
94  {
95  case LSBFirst:
96  tmp = iConvertImage( ilutCurImage,IL_BGR,IL_UNSIGNED_BYTE );
97  break;
98  case MSBFirst:
99  tmp = iConvertImage( ilutCurImage,IL_RGB,IL_UNSIGNED_BYTE );
100  break;
101  default:
102  return;
103  }
104 
105  if ( !tmp ) return;
106  iXGrabImage( tmp );
107 
108 
109  switch ( img->format )
110  {
111  case ZPixmap:
112 
113  for ( y = 0; y < height; y ++ )
114  {
115  dY = y * img->bytes_per_line;
116  sY = y * grain;
117 
118  for ( x = 0; x < width; x ++ )
119  {
120  dX = x * img->bits_per_pixel / 8;
121  sX = x * bytes;
122 
123  for ( z = 0; z < bytes; z ++ )
124  {
125  img->data[dX+dY+z] = data[sX+sY+z];
126  }
127  }
128  }
129 
130  break;
131 
132  case XYPixmap:
133 
134  for ( y = 0; y < height; y ++ )
135  {
136  sY = y * grain;
137 
138  for ( x = 0; x < width; x ++ )
139  {
140  sX = x * bytes;
141 
142  for ( z = 0; z < bits; z ++ )
143  {
144  sZ = z / 8;
145  dZ = z % 8;
146 
147  if ( data[sY+sX+sZ] & ( 1 << dZ ) )
148  {
149  plane = bits - z - 1;
150 
151  sZ = x % 8;
152  dX = x / 8;
153  dY = y * img->bytes_per_line;
154  dZ = plane * img->bytes_per_line * height;
155 
156  img->data[dZ+dY+dX] |= 1 << sZ;
157  }
158  }
159  }
160  }
161 
162  break;
163 
164  default:
165  ilSetError( ILUT_NOT_SUPPORTED );
166  }
167 
168  ilCloseImage( tmp );
169 }

273 XImage * ILAPIENTRY ilutXShmCreateImage( Display * dpy, XShmSegmentInfo * info )
274 {
275  Visual * vis;
276  XImage * img;
277 
278  // Get server supported format
279 
280  int size,format = XShmPixmapFormat( dpy );
281 
282  // Grab the current image
283 
284  if (!iXGrabCurrentImage()) {
285  return NULL;
286  }
287 
288  // Create a shared image
289 
290  vis = CopyFromParent;
291  img = XShmCreateImage( dpy,vis, 24,format,NULL,info,width,height );
292 
293  if (!img) {
294  return NULL;
295  }
296 
297  // Create shared memory
298 
299  size = img->bytes_per_line * img->height;
300 
301  info->shmid = shmget( IPC_PRIVATE, size, IPC_CREAT | 0666 );
302  info->shmaddr = img->data = shmat( info->shmid, 0, 0 );
303  info->readOnly = False;
304 
305  // Attach to server
306 
307  XShmAttach( dpy,info );
308 
309  // Copy image pixels to shared memory
310 
311  iXConvertImage( dpy,img );
312 
313  return img;
314 }
