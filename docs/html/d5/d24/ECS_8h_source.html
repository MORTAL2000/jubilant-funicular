267  if (!does_entity_exist(entity)) return utils::make_none<ComponentID>();

269  if (list.size() <= entity.idx) list.reserve(entity.idx+1);
270 
271  if (!list.insert_emplace(entity, std::forward<Args>(args)...)) {
272  NTA_ASSERT(false, "This should never happen 1");
273  }
274  T& cmpn = list[entity].unwrap();
275  cmpn.m_ecs = this;
276  cmpn.m_id = m_cmpn_gen();

278  m_component_info.reserve(cmpn.m_id.idx+1);

280  if (!m_component_info.insert(cmpn, info)) {
