248  utils::Option<ComponentID> ECS::add_component(Entity entity, Args&&... args) {
249  if (!does_entity_exist(entity)) return utils::make_none<ComponentID>();
250  ComponentList<T>& list = m_components.find<ComponentList<T>>();
251  list.reserve(entity.idx+1);
252 
253  if (!list.insert_emplace(entity, std::forward<Args>(args)...)) {
254  return utils::make_none<ComponentID>();
255  }
256  T& cmpn = list[entity].unwrap();
257  cmpn.m_id = m_cmpn_gen();
258 
259  m_component_info.reserve(cmpn.m_id.idx+1);
260  ComponentInfo info = { .type = typeid(T).hash_code(), .owner = entity };
261  if (!m_component_info.insert(cmpn.m_id, info)) {
262  assert(false && "This should never happen");
263  }
264  return utils::make_some(cmpn.m_id);
265  }
