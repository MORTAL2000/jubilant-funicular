24 ILint ILAPIENTRY iSizeSeek(ILint Offset, ILuint Mode)
25 {
26  switch (Mode)

28  case IL_SEEK_SET:
29  CurPos = Offset;
30  if (CurPos > MaxPos)

33 
34  case IL_SEEK_CUR:

37 
38  case IL_SEEK_END:
39  CurPos = MaxPos + Offset; // Offset should be negative in this case.
40  break;
41 
42  default:
43  ilSetError(IL_INTERNAL_ERROR); // Should never happen!
44  return -1; // Error code
45  }
46

48  MaxPos = CurPos;

51 }
