137 int parse_function(const char * string, char * name, char * params)
138 {
139  int i;
140  /* num of whitespaces in front of the function */
141  int num_front_whitespaces = 0;
142  /* num of character that bears the first parameter char - beyond '(' */
143  int in_parameters = 0;
144  /* trim the leading whitespaces */
145  for (i = 0; i < long_strlen - 1; i++)
146  if(string[i] == ' ' || string[i] == '\t')
147  num_front_whitespaces++;
148  else
149  break;
150  /* copy the function name */
151  for (; i < long_strlen - 1 && string[i] != '('; i++)
152  if(string[i] == ' ' || string[i] == '\t')
153  break;
154  else
155  name[i - num_front_whitespaces] = string[i];
156  /* terminate the string */
157  name[i - num_front_whitespaces] = '\0';
158  /* finally get the parameter */
159  for (; i < long_strlen - 1 && string[i] != ')'; i++)
160  if(string[i] == '(')
161  {
162  in_parameters = i + 1;
163  continue;
164  }
165  else
166  {
167  if(in_parameters != 0)
168  params[i - in_parameters] = string[i];
169  }
170  /* again terminate the string */
171  params[i - in_parameters] = '\0';
172  return 0;
173 }

179 int remove_whitespaces(const char * string, char * nonwhitespaced)
180 {
181  strncpy(nonwhitespaced, string, strlen(string) + 1);
182  int i;
183  /* How many whitespace chars were skipped? */
184  int num_whitespaced = 0;
185  /* Go from the beginning and don't stop until the either end */
186  for (i = 0; string[i] != '\0' && i < long_strlen; i++)
187  /* Ho, a whitespace that shouldn't be missed! */
188  if( string[i] == ' ' || string[i] == '\t' )
189  {
190  num_whitespaced++;
191  continue;
192  }
193  else /* No whitespace, moreover we are already inside the string... */
194  {
195  /* Here we copy the stuff to the output parameter */
196  nonwhitespaced[i - num_whitespaced] = string[i];
197  }
198 }

202 int parse_arguments(int argc, const char * argv[], Params * parameters)
203 {
204  /* How many ILU functions are we going to apply to the image? */
205  int calls_count = 0;
206  /* Let's store their corresponding indexes in argv... */
207  int * fun_to_call = (int *)malloc(argc / 2 * sizeof(int));
208 
209  int i;
210  for (i = 1; i < argc; i++)
211  {
212  /* Find out what to do and what to expect */
213  if (argv[i][0] == '-')
214  {/* Ho, an option was passed... */
215  if (argv[i][1] == '-')
216  {/* Deal with long options */
217  if(strncmp(argv[i], "--apply", long_strlen))
218  goto apply; /* Yeah, there are GOTO's here :-) */
219  else if(strncmp(argv[i], "--load-from", long_strlen))
220  goto load_from;
221  else if(strncmp(argv[i], "--save-to", long_strlen))
222  goto save_to;
223  else if(strncmp(argv[i], "--verbose", long_strlen))
224  goto verbose;
225  }
226  else switch(argv[i][1]) /* Well, maybe it wasn't a long option :-) */
227  {/* Deal with long options */
228  case 'h':
229  case '?':
230  help:
231  parameters->Flags |= FLAG_HELP;
232  break;
233  case 'a':
234  apply:
235  if (argc > i + 1) /* that there is maybe something like the parameter out there... */
236  fun_to_call[calls_count++] = i + 1;
237  break;
238  case 'l':
239  load_from:
240  if (argc > i + 1)
241  {/* that there is maybe something like the parameter out there... */
242  strncpy(parameters->Load_filename, argv[i + 1], long_strlen);
243  parameters->Flags |= FLAG_LOAD;
244  }
245  break;
246  case 's':
247  save_to:
248  if (argc > i + 1)
249  {/* that there is maybe something like the parameter out there... */
250  strncpy(parameters->Save_filename, argv[i + 1], long_strlen);
251  parameters->Flags |= FLAG_SAVE;
252  }
253  break;
254  case 'v':
255  verbose:
256  parameters->Flags |= FLAG_VERBOSE;
257  break;
258  }
259  /* We don't use complicated parameters here, so no worries */
260  }/* endif (argv[i][0] == '-') */
261  }/* endif (argv[i][0] == '-') */
262  /* let's save the valuable info to the output structure... */
263  parameters->Calls_count = calls_count;
264  /* and let's also store the calls as passed by the user */
265  parameters->Calls_strings = (char **)malloc(parameters->Calls_count * sizeof (char *));
266  for (i = 0; i < calls_count; i++)
267  {
268  /* Yeah, there is probably more memory allocated than needed... */
269  parameters->Calls_strings[i] = (char *)malloc(sizeof(char) * long_strlen);
270  strncpy(parameters->Calls_strings[i], argv[ fun_to_call[i] ], long_strlen);
271  }
272 
273  /* clean the mess... */
274  free(fun_to_call);
275  fun_to_call = 0;
276 }

300 int perform_operation(const char * operation, int verbose)
301 {
302  /* Where to store the first parsing results? */
303  char function[long_strlen], params[long_strlen], solid_params[long_strlen];
304  /* Get the function name string and parameters string */
305  parse_function(operation, function, params);
306  /* Get rid of any whitespaces from the parameters string */
307  remove_whitespaces(params, solid_params);
308  if (verbose)
309  printf("Calling %s(%s)\n", function, solid_params);
310  /* What function was wanted? -1 means that we don't know */
311  int function_index = -1;
312  int i;
313  for (i = 0; i < ILU_FUN_COUNT; i++)
314  if (strncmp(function, ilu_functions[i].Name, short_strlen) == 0)
315  {/* Yeah, this function was wanted. Let's have its index from the ilu_functions array */
316  function_index = i;
317  break; /* nothing to do here any more */
318  }
319  if (function_index == -1)
320  {/* Seems we haven't found anything... */
321  fprintf(stderr, "Error: You have specified an invalid function name '%s' (have you called %s).\nRun '%s --help' command to get some help\n", function, operation, program_name);
322  return 1;
323  }
324  /* We are going to try something and we want to know how it ended */
325  ILboolean return_value;
326  switch (ilu_functions[function_index].Parameter_type)
327  {/* First semi-automatic processing according to type of parameters */
328  case PARAM_VOID:
329  {
330  ILboolean (* function)() = ilu_functions[function_index].Callback;
331  return_value = function();
332  break;
333  }/* endcase PARAM_VOID */
334  case PARAM_ILUINT:
335  {
336  /* first assign and determine the type of the Callback */
337  ILboolean (* function)(ILuint) = ilu_functions[function_index].Callback;
338  /* then declare the parameter variables */
339  ILuint param_value;
340  /* fill them */
341  int success = sscanf(solid_params, "%u", & param_value);
342  if (success != 1)
343  {/* see how it ended */
344  fprintf(stderr, "Error interpreting '%s' as unsigned integer (when calling %s)\n", solid_params, operation);
345  break;
346  }
347  /* execute the command and store the result */
348  return_value = function(param_value);
349  break;
350  }/* endcase PARAM_ILUINT */
351  case PARAM_ILFLOAT:
352  {
353  ILboolean (* function)(ILfloat) = ilu_functions[function_index].Callback;
354  double param_value;
355  int success = sscanf(solid_params, "%lf", & param_value);
356  if (success != 1)
357  {
358  fprintf(stderr, "Error interpreting '%s' as float (when calling %s)\n", solid_params, operation);
359  break;
360  }
361  return_value = function((ILfloat)param_value);
362  break;
363  }/* endcase PARAM_ILFLOAT */
364  case PARAM_OTHERS:
365  switch (function_index)
366  {/* next, the manual processing according to names */
367  case ILU_SHARPEN:
368  {
369  ILboolean (* function)(ILfloat, ILuint) = ilu_functions[function_index].Callback;
370  double factor;
371  ILuint iter;
372  int success = sscanf(solid_params, "%lf,%u", & factor, & iter);
373  if (success != 2)
374  {
375  fprintf(stderr, "Error interpreting '%s' as floating-point number and unsigned integer separated by comma (when calling %s)\n", solid_params, operation);
376  break;
377  }
378  return_value = function((ILfloat)factor, iter);
379  break;
380  }/* endcase ILU_SHARPEN */
381  case ILU_CROP:
382  {
383  ILboolean (* function)(ILuint, ILuint, ILuint, ILuint, ILuint, ILuint ) = ilu_functions[function_index].Callback;
384  ILuint xoff, yoff, zoff, width, height, depth;
385  int success = sscanf(solid_params, "%u,%u,%u,%u,%u,%u", & xoff, & yoff, & zoff, & width, & height, & depth);
386  if (success != 6)
387  {
388  fprintf(stderr, "Error interpreting '%s' as 6 unsigned integers separated by comma (when calling %s)\n", solid_params, operation);
389  break;
390  }
391  return_value = function(xoff, yoff, zoff, width, height, depth);
392  break;
393  }/* endcase ILU_CROP */
394  case ILU_ENLARGECANVAS:
395  case ILU_SCALE:
396  {
397  ILboolean (* function)(ILuint, ILuint, ILuint) = ilu_functions[function_index].Callback;
398  ILuint width, height, depth;
399  int success = sscanf(solid_params, "%u,%u,%u", & width, & height, & depth);
400  if (success != 3)
401  {
402  fprintf(stderr, "Error interpreting '%s' as 3 unsigned integers separated by comma (when calling %s)\n", solid_params, operation);
403  break;
404  }
405  return_value = function(width, height, depth);
406  break;
407  }/* endcase ILU_ENLARGECANVAS + ILU_SCALE */
408  case ILU_ENLARGEIMAGE:
409  case ILU_SCALECOLOURS:
410  {
411  ILboolean (* function)(ILfloat, ILfloat, ILfloat) = ilu_functions[function_index].Callback;
412  double first, second, third;
413  int success = sscanf(solid_params, "%lf,%lf,%lf", & first, & second, & third);
414  if (success != 3)
415  {
416  fprintf(stderr, "Error interpreting '%s' as 3 floating-point numbers separated by comma (when calling %s)\n", solid_params, operation);
417  break;
418  }
419  return_value = function((ILfloat)first, (ILfloat)second, (ILfloat)third);
420  break;
421  }/* endcase ILU_ENLARGEIMAGE + ILU_SCALECOLOURS */
422  case ILU_ROTATE3D:
423  case ILU_SATURATE4F:
424  {
425  ILboolean (* function)(ILfloat, ILfloat, ILfloat, ILfloat) = ilu_functions[function_index].Callback;
426  double first, second, third, fourth;
427  int success = sscanf(solid_params, "%lf,%lf,%lf,%lf", & first, & second, & third, & fourth);
428  if (success != 4)
429  {
430  fprintf(stderr, "Error interpreting '%s' as 4 floating-point numbers separated by comma (when calling %s)\n", solid_params, operation);
431  break;
432  }
433  return_value = function((ILfloat)first, (ILfloat)second, (ILfloat)third, (ILfloat)fourth);
434  break;
435  }/* endcase ILU_ROTATE3D + ILU_SATURATE4F */
436  case ILU_REPLACECOLOUR:
437  {
438  ILboolean (* function)(ILubyte, ILubyte, ILubyte, ILfloat ) = ilu_functions[function_index].Callback;
439  ILuint red, green, blue;
440  double tolerance;
441  int success = sscanf(solid_params, "%u,%u,%u,%lf", & red, & green, & blue, & tolerance);
442  if (success != 4)
443  {
444  fprintf(stderr, "Error interpreting '%s' as 3 8-bit unsigned integers and one floating-point number separated by comma (when calling %s)\n", solid_params, operation);
445  break;
446  }
447  return_value = function((ILubyte)red, (ILubyte)green, (ILubyte)blue, (ILfloat)tolerance);
448  break;
449  }/* endcase ILU_ROTATE3D + ILU_SATURATE4F * */
450  /* iluConvolution(ILint *matrix, ILint scale, ILint bias); Any idea about this? */
451  }/* endswitch(function_index) */
452  break;
453 
454  }/* endswitch (ilu_functions[function_index].Parameter_type) */
455  /* It didn't end good for some reason... */
456  if (return_value == IL_FALSE)
457  {
458  int error= ilGetError();
459  fprintf(stderr, "Something got wrong when calling %s(%s): %s\n", function, solid_params, iluErrorString(error) );
460  return error;
461  }
462  return 0;
463 }

465 int do_stuff(const Params * parameters)
466 {
467  if (parameters->Flags & FLAG_HELP || ((parameters->Flags | FLAG_LOAD | FLAG_SAVE) != parameters->Flags) )
468  {/* We wanted HELP or we did not get SAVE or LOAD */
469  print_help(); /* tell the loser what to do, then :-) */
470  return 0;
471  }
472  int verbose = parameters->Flags & FLAG_VERBOSE;
473 
474  int image_handle;
475  int w, h;
476  ILboolean result;
477 
478  /* Quite obvious stuff, just load an image */
479  ilGenImages(1, & image_handle);
480  ilBindImage(image_handle);
481  result = ilLoadImage(parameters->Load_filename);
482  if (result == IL_FALSE)
483  {
484  int error = ilGetError();
485  fprintf(stderr, "Error: Something went wrong when loading file '%s' (%s)\n", parameters->Load_filename, iluErrorString(error));
486  return error;
487  }
488  /* If we get image's dimensions, people will believe that we have actually loaded something :-) */
489  w = ilGetInteger(IL_IMAGE_WIDTH);
490  h = ilGetInteger(IL_IMAGE_HEIGHT);
491  if (verbose)
492  printf("Loaded '%s', size %dx%d\n", parameters->Load_filename, w, h);
493  /* Now let's do our stuff!!! */
494  int i;
495  for (i = 0; i < parameters->Calls_count; i++)
496  perform_operation(parameters->Calls_strings[i], verbose);
497  /* our stuff has been done... */
498 
499  result = ilSaveImage(parameters->Save_filename);
500  if (result == IL_FALSE)
501  {
502  int error = ilGetError();
503  fprintf(stderr, "Error: Something went wrong when saving file '%s' (%s)\n", parameters->Save_filename, iluErrorString(error));
504  ilDeleteImages(1, & image_handle);
505  return error;
506  }
507  ilDeleteImages(1, & image_handle);
508  return 0;
509 }

511 int main(int argc, const char * argv[])
512 {
513  /* Name our little program */
514  strncpy(program_name, argv[0], long_strlen);
515  /* Prepare ilu functions "database" */
516  init_strings();
