51 SDL_Surface *ILAPIENTRY ilutConvertToSDLSurface(unsigned int flags)
52 {
53  SDL_Surface *Bitmap = NULL;
54  ILuint i = 0, Pad, BppPal;
55  ILubyte *Dest, *Data;
56  ILimage *Image;
57 
58  Image = ilutCurImage = ilGetCurImage();
59  if (ilutCurImage == NULL) {
60  ilSetError(ILUT_ILLEGAL_OPERATION);
61  return NULL;
62  }
63 
64  InitSDL();
65 
66  // Should be IL_BGR(A).
67  if (ilutCurImage->Format == IL_RGB || ilutCurImage->Format == IL_RGBA) {
68  if (!isBigEndian) {
69  //iluSwapColours(); // No need to swap colors. Just use the bitmasks.
70  rmask = 0x00FF0000;
71  gmask = 0x0000FF00;
72  bmask = 0x000000FF;
73  }
74  }
75  else if (ilutCurImage->Format == IL_BGR || ilutCurImage->Format == IL_BGRA) {
76  if (isBigEndian) {
77  rmask = 0x0000FF00;
78  gmask = 0x00FF0000;
79  bmask = 0xFF000000;
80  }
81  }
82  else if (Image->Format != IL_COLOR_INDEX) { // We have to convert the image.
83  #ifdef __BIG_ENDIAN__
84  Image = iConvertImage(Image, IL_RGBA, IL_UNSIGNED_BYTE);
85  #else
86  Image = iConvertImage(Image, IL_BGRA, IL_UNSIGNED_BYTE);
87  #endif
88  if (Image == NULL)
89  return NULL;
90  }
91 
92  if (Image->Type != IL_UNSIGNED_BYTE) {
93  // We do not have to worry about Image != iCurImage at this point, because if it was converted,
94  // it was converted to a type of unsigned byte.
95  Image = iConvertImage(Image, Image->Format, IL_UNSIGNED_BYTE);
96  if (Image == NULL)
97  return NULL;
98  }
99 
100  Data = Image->Data;
101  if (Image->Origin == IL_ORIGIN_LOWER_LEFT) {
102  Data = iGetFlipped(Image);
103  if (Data == NULL)
104  goto done;
105  }
106 
107  Bitmap = SDL_CreateRGBSurface(flags, Image->Width, Image->Height, Image->Bpp * 8,
108  rmask,gmask,bmask,amask);
109  if (Bitmap == NULL)
110  goto done;
111 
112  if (SDL_MUSTLOCK(Bitmap))
113  SDL_LockSurface(Bitmap);
114 
115  Pad = Bitmap->pitch - Image->Bps;
116  if (Pad == 0) {
117  memcpy(Bitmap->pixels, Data, Image->SizeOfData);
118  }
119  else { // Must pad the lines on some images.
120  Dest = Bitmap->pixels;
121  for (i = 0; i < Image->Height; i++) {
122  memcpy(Dest, Data + i * Image->Bps, Image->Bps);
123  imemclear(Dest + Image->Bps, Pad);
124  Dest += Bitmap->pitch;
125  }
126  }
127 
128  if (SDL_MUSTLOCK(Bitmap))
129  SDL_UnlockSurface(Bitmap);
130 
131  if (Image->Format == IL_COLOR_INDEX) {
132  BppPal = ilGetBppPal(Image->Pal.PalType);
133  switch (ilutCurImage->Pal.PalType)
134  {
135  case IL_PAL_RGB24:
136  case IL_PAL_RGB32:
137  case IL_PAL_RGBA32:
138  for (i = 0; i < ilutCurImage->Pal.PalSize / BppPal; i++) {
139  (Bitmap->format)->palette->colors[i].r = ilutCurImage->Pal.Palette[i*BppPal+0];
140  (Bitmap->format)->palette->colors[i].g = ilutCurImage->Pal.Palette[i*BppPal+1];
141  (Bitmap->format)->palette->colors[i].b = ilutCurImage->Pal.Palette[i*BppPal+2];
142  (Bitmap->format)->palette->colors[i].unused = 0xFF;
143  }
144  break;
145  case IL_PAL_BGR24:
146  case IL_PAL_BGR32:
147  case IL_PAL_BGRA32:
148  for (i = 0; i < ilutCurImage->Pal.PalSize / BppPal; i++) {
149  (Bitmap->format)->palette->colors[i].b = ilutCurImage->Pal.Palette[i*BppPal+0];
150  (Bitmap->format)->palette->colors[i].g = ilutCurImage->Pal.Palette[i*BppPal+1];
151  (Bitmap->format)->palette->colors[i].r = ilutCurImage->Pal.Palette[i*BppPal+2];
152  (Bitmap->format)->palette->colors[i].unused = 0xFF;
153  }
154  break;
155  default:
156  ilSetError(IL_INTERNAL_ERROR); // Do anything else?
157  }
158  }
159 
160 done:
161  if (Data != Image->Data)
162  ifree(Data); // This is flipped data.
163  if (Image != ilutCurImage)
164  ilCloseImage(Image); // This is a converted image.
165  return Bitmap; // This is NULL if there was an error.
166 }
