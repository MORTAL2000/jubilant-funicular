61 class PhysicsComponent : public nta::Component {
62 private:
63  vec2 m_pos;
64  vec2 m_vel;
65  vec2 m_acc;
66  float m_rad;
67 public:
68  PhysicsComponent() {
69  m_pos = m_vel = m_acc = vec2(0);
70  m_rad = 5;
71  }
72  PhysicsComponent(nta::crvec2 pos, float radius) : m_pos(pos), m_rad(radius) {
73  m_vel = m_acc = vec2(0);
74  }
75 
76  float get_mass() const { return m_rad*m_rad*BALL_DENSITY; }
77  vec2 get_pos() const { return m_pos; }
78  void step(float dt) {
79  vec2 acc = m_acc + GRAVITY;
80  m_vel += acc * dt;
81  m_pos += m_vel * dt;
82 
83  if (m_pos.y - m_rad < GROUND_Y) {
84  m_vel.y = abs(m_vel.y);
85  }
86 
87  send(nta::Message(0, &m_pos));
88  }
89  // Reminder: Ignore the physics in this example
90  // It's not the focus and it is wrong
91  void collide(PhysicsComponent& other, float dt) {
92  vec2 diff = other.m_pos - m_pos;
93  float dist2 = diff.x*diff.x + diff.y*diff.y;
94 
95  float rad_sum = m_rad + other.m_rad;
96  if (dist2 < rad_sum * rad_sum) {
97  float tot_mass = get_mass() + other.get_mass();
98  float mass_diff = get_mass() - other.get_mass();
99  vec2 vel_diff = m_vel - other.m_vel;
100 
101  vec2 v1 = 2*other.get_mass()*dot(vel_diff, diff)*diff;
102  v1 = m_vel - v1/(tot_mass*dist2);
103 
104  vec2 v2 = 2*get_mass()*dot(vel_diff, diff)*diff;
105  v2 = other.m_vel + v2/(tot_mass*dist2);
106 
107  float decay = nta::Random::randFloat(COLLISION_DECAY_MIN, COLLISION_DECAY_MAX);
108  m_vel = v1 * decay;
109  other.m_vel = v2 * decay;
110 
111  float overlap = rad_sum - sqrt(dist2);
112  m_pos -= diff*overlap/rad_sum;
113  }
114  }
115  // Every Component needs to be able to receive Messages
116  void receive(const nta::Message& _) {}
117 };

192 void MainScreen::update() {
193  // Adds a ball whenver you click on the screen
194  if (nta::InputManager::justPressed(SDL_BUTTON_LEFT)) {
195  addBall(getMouse());
196  }
197 
198  vector<PhysicsComponent*> phys_cpmns = m_system.get_component_list<PhysicsComponent>();
199  for (int i = 0; i < phys_cpmns.size(); i++) {
200  for (int j = i+1; j < phys_cpmns.size(); j++) {
201  phys_cpmns[i]->collide(*phys_cpmns[j], 1./60);
202  }
203  }
204  for (PhysicsComponent* cmpn : phys_cpmns) {
205  cmpn->step(1./60);
206  }
207 
208  // removes any ball whose center goes off the screen
209  for (int i = 0; i < m_balls.size(); i++) {
210  auto& ball = m_balls[i];
211  // Get the ball's position from its physics component
212  vec2 pos = m_system.get_component<PhysicsComponent>(ball).get_pos();
213  if (pos.x > 100 || pos.x < -100 || pos.y > 100 || pos.y < -100) {
214  m_system.delete_entity(ball);
215  // ordering in m_balls doesn't matter so might as well
216  // do an O(1) deletion
217  swap(ball, m_balls.back());
218  m_balls.pop_back();
219  i--;
220  }
221  }
222  // Note: there are better ways to do the above
223  // * one would be to make PhysicsComponent an nta::Subject
224  // * and make MainScreen an nta::Observer. Then, the component could signal
225  // * MainScreen of a ball going out of bounds during the call to step.
226  // * MainScreen could keep track of all the balls then went out of bounds
227  // * and delete them all at the end of the update method
228 
229  if (m_balls.size() == 0) {
230  // Let the ScreenManager know that it is time to switch screens
231  // Switch to the screen at index -1 (i.e. exit the ScreenManager)
232  m_state = nta::ScreenState::SWITCH;
233  m_nextIndex = -1;
234  }
235 }
