72 // Internal function used to load the ROT.
73 ILboolean iLoadRotInternal(void)
74 {
75  ILubyte Form[4], FormName[4];
76  ILuint FormLen, Width, Height, Format, Channels, CompSize;
77  ILuint MipSize, MipLevel, MipWidth, MipHeight;
78  ILenum FormatIL;
79  ILimage *Image;
80  ILboolean BaseCreated = IL_FALSE;
81  ILubyte *CompData = NULL;
82 
83  if (iCurImage == NULL) {
84  ilSetError(IL_ILLEGAL_OPERATION);
85  return IL_FALSE;
86  }
87 
88  // The first entry in the file must be 'FORM', 0x20 in a big endian integer and then 'HEAD'.
89  iread(Form, 1, 4);
90  FormLen = GetBigUInt();
91  iread(FormName, 1, 4);
92  if (strncmp((char*)Form, "FORM", 4) || FormLen != 0x14 || strncmp((char*)FormName, "HEAD", 4)) {
93  ilSetError(IL_INVALID_FILE_HEADER);
94  return IL_FALSE;
95  }
96 
97  // Next follows the width, height and format in the header.
98  Width = GetLittleUInt();
99  Height = GetLittleUInt();
100  Format = GetLittleUInt();
101 
102  //@TODO: More formats.
103  switch (Format)
104  {
105  case ROT_RGBA32: // 32-bit RGBA format
106  Channels = 4;
107  FormatIL = IL_RGBA;
108  break;
109 
110  case ROT_DXT1: // DXT1 (no alpha)
111  Channels = 4;
112  FormatIL = IL_RGBA;
113  break;
114 
115  case ROT_DXT3: // DXT3
116  case ROT_DXT5: // DXT5
117  Channels = 4;
118  FormatIL = IL_RGBA;
119  // Allocates the maximum needed (the first width/height given in the file).
120  CompSize = ((Width + 3) / 4) * ((Height + 3) / 4) * 16;
121  CompData = (ILubyte*)ialloc(CompSize);
122  if (CompData == NULL)
123  return IL_FALSE;
124  break;
125 
126  default:
127  ilSetError(IL_INVALID_FILE_HEADER);
128  return IL_FALSE;
129  }
130 
131  if (Width == 0 || Height == 0) {
132  ilSetError(IL_INVALID_FILE_HEADER);
133  return IL_FALSE;
134  }
135 
136  //@TODO: Find out what this is.
137  GetLittleUInt(); // Skip this for the moment. This appears to be the number of channels.
138 
139  // Next comes 'FORM', a length and 'MIPS'.
140  iread(Form, 1, 4);
141  FormLen = GetBigUInt();
142  iread(FormName, 1, 4);
143  //@TODO: Not sure if the FormLen has to be anything specific here.
144  if (strncmp((char*)Form, "FORM", 4) || strncmp((char*)FormName, "MIPS", 4)) {
145  ilSetError(IL_INVALID_FILE_HEADER);
146  return IL_FALSE;
147  }
148 
149  //@TODO: Can these mipmap levels be in any order? Some things may be easier if the answer is no.
150  Image = iCurImage;
151  do {
152  // Then we have 'FORM' again.
153  iread(Form, 1, 4);
154  // This is the size of the mipmap data.
155  MipSize = GetBigUInt();
156  iread(FormName, 1, 4);
157  if (strncmp((char*)Form, "FORM", 4)) {
158  if (!BaseCreated) { // Our file is malformed.
159  ilSetError(IL_INVALID_FILE_HEADER);
160  return IL_FALSE;
161  }
162  // We have reached the end of the mipmap data.
163  break;
164  }
165  if (strncmp((char*)FormName, "MLVL", 4)) {
166  ilSetError(IL_INVALID_FILE_HEADER);
167  return IL_FALSE;
168  }
169 
170  // Next is the mipmap attributes (level number, width, height and length)
171  MipLevel = GetLittleUInt();
172  MipWidth = GetLittleUInt();
173  MipHeight = GetLittleUInt();
174  MipSize = GetLittleUInt(); // This is the same as the previous size listed -20 (for attributes).
175 
176  // Lower level mipmaps cannot be larger than the main image.
177  if (MipWidth > Width || MipHeight > Height || MipSize > CompSize) {
178  ilSetError(IL_INVALID_FILE_HEADER);
179  return IL_FALSE;
180  }
181 
182  // Just create our images here.
183  if (!BaseCreated) {
184  if (!ilTexImage(MipWidth, MipHeight, 1, Channels, FormatIL, IL_UNSIGNED_BYTE, NULL))
185  return IL_FALSE;
186  BaseCreated = IL_TRUE;
187  }
188  else {
189  Image->Mipmaps = ilNewImageFull(MipWidth, MipHeight, 1, Channels, FormatIL, IL_UNSIGNED_BYTE, NULL);
190  Image = Image->Mipmaps;
191  }
192 
193  switch (Format)
194  {
195  case ROT_RGBA32: // 32-bit RGBA format
196  if (iread(Image->Data, Image->SizeOfData, 1) != 1)
197  return IL_FALSE;
198  break;
199 
200  case ROT_DXT1:
201  // Allocates the size of the compressed data.
202  CompSize = ((MipWidth + 3) / 4) * ((MipHeight + 3) / 4) * 8;
203  if (CompSize != MipSize) {
204  ilSetError(IL_INVALID_FILE_HEADER);
205  return IL_FALSE;
206  }
207  CompData = (ILubyte*)ialloc(CompSize);
208  if (CompData == NULL)
209  return IL_FALSE;
210 
211  // Read in the DXT1 data...
212  if (iread(CompData, CompSize, 1) != 1)
213  return IL_FALSE;
214  // ...and decompress it.
215  if (!DecompressDXT1(Image, CompData)) {
216  ifree(CompData);
217  return IL_FALSE;
218  }
219  if (ilGetInteger(IL_KEEP_DXTC_DATA) == IL_TRUE) {
220  Image->DxtcSize = CompSize;
221  Image->DxtcData = CompData;
222  Image->DxtcFormat = IL_DXT1;
223  CompData = NULL;
224  }
225  break;
226 
227  case ROT_DXT3:
228  // Allocates the size of the compressed data.
229  CompSize = ((MipWidth + 3) / 4) * ((MipHeight + 3) / 4) * 16;
230  if (CompSize != MipSize) {
231  ilSetError(IL_INVALID_FILE_HEADER);
232  return IL_FALSE;
233  }
234  CompData = (ILubyte*)ialloc(CompSize);
235  if (CompData == NULL)
236  return IL_FALSE;
237 
238  // Read in the DXT3 data...
239  if (iread(CompData, MipSize, 1) != 1)
240  return IL_FALSE;
241  // ...and decompress it.
242  if (!DecompressDXT3(Image, CompData)) {
243  ifree(CompData);
244  return IL_FALSE;
245  }
246  if (ilGetInteger(IL_KEEP_DXTC_DATA) == IL_TRUE) {
247  Image->DxtcSize = CompSize;
248  Image->DxtcData = CompData;
249  Image->DxtcFormat = IL_DXT3;
250  CompData = NULL;
251  }
252  break;
253 
254  case ROT_DXT5:
255  // Allocates the size of the compressed data.
256  CompSize = ((MipWidth + 3) / 4) * ((MipHeight + 3) / 4) * 16;
257  if (CompSize != MipSize) {
258  ilSetError(IL_INVALID_FILE_HEADER);
259  return IL_FALSE;
260  }
261  CompData = (ILubyte*)ialloc(CompSize);
262  if (CompData == NULL)
263  return IL_FALSE;
264 
265  // Read in the DXT5 data...
266  if (iread(CompData, MipSize, 1) != 1)
267  return IL_FALSE;
268  // ...and decompress it.
269  if (!DecompressDXT5(Image, CompData)) {
270  ifree(CompData);
271  return IL_FALSE;
272  }
273  // Keeps a copy
274  if (ilGetInteger(IL_KEEP_DXTC_DATA) == IL_TRUE) {
275  Image->DxtcSize = CompSize;
276  Image->DxtcData = CompData;
277  Image->DxtcFormat = IL_DXT5;
278  CompData = NULL;
279  }
280  break;
281  }
282  ifree(CompData); // Free it if it was not saved.
283  } while (!ieof()); //@TODO: Is there any other condition that should end this?
284 
285  return ilFixImage();
286 }
