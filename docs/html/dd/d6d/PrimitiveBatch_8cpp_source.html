93  numVertices += p->vertices.size();

95  Vertex2D vertexData[numVertices];

97  toPrimitiveType(m_primitives[0]->vertices.size()));
98  int cv = 0; // current vertex
99  for (int i = 0; i < m_primitives[0]->vertices.size(); i++) {
100  vertexData[cv++] = m_primitives[0]->vertices[i];
101  }
102  int offset = m_primitives[0]->vertices.size();
103  for (int cp = 1; cp < m_primitives.size(); cp++) { // current primitive
104  if (m_primitives[cp]->textureID != m_primitives[cp-1]->textureID ||
105  m_primitives[cp]->vertices.size() != m_primitives[cp-1]->vertices.size()) {
106  m_renderBatches.emplace_back(m_primitives[cp]->textureID, offset,
107  m_primitives[cp]->vertices.size(),
108  toPrimitiveType(m_primitives[cp]->vertices.size()));
109  } else {
110  m_renderBatches.back().numVertices += m_primitives[cp]->vertices.size();
111  }
112  for (int i = 0; i < m_primitives[cp]->vertices.size(); i++) {
113  vertexData[cv++] = m_primitives[cp]->vertices[i];
114  }
115  offset += m_primitives[cp]->vertices.size();
116  }

118  glBufferData(GL_ARRAY_BUFFER, sizeof(vertexData), nullptr, GL_DYNAMIC_DRAW);
