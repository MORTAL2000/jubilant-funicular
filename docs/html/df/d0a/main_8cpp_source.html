40 void SquareScreen::init() {
41  nta::Logger::writeToLog("Initializing main screen...");
42 
43  // Admitedlly, the simple2D I'm using is more complicated than need be
44  // Loads a GLSLProgram (vertex and fragment shader) from a pair of files
45  // named "simple2D.vert" and "simple2D.frag"
46  m_simple_prog = nta::SystemManager::getGLSLProgram("simple2D");
47  if (!m_simple_prog->isLinked()) {
48  // Adds the various attributes used by simple2D.vert
49  // the order here must match the order of the fields of Vertex2D
50  m_simple_prog->addAttribute("pos");
51  m_simple_prog->addAttribute("color");
52  m_simple_prog->addAttribute("uv");
53  m_simple_prog->addAttribute("hasTexture");
54  m_simple_prog->linkShaders();
55  }
56  m_simple_prog->use();
57  // bind GL_TEXTURE0 by default
58  glUniform1i(m_simple_prog->getUniformLocation("sampler"), 0);
59  m_simple_prog->unuse();

76 void SquareScreen::render() {
77  glClearColor(0, 0, 0, 0);
78  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
79 
80  // Collect all the glyphs to be drawn
81  m_batch.begin();
82  // Draws a square whose topleft corner is at (-25, 25) and whose
83  // width and height are both 50. The uv (i.e. texture) coordinates
84  // used are given by a square whose topleft corner is at (0, 0) and
85  // whose width and height are both 0. This is an empty square which
86  // would be problematic except that the texture id suuplied here is
87  // 0, so this glyph won't actually use a texture at all (i.e. since
88  // the third parameter is 0, the second one is irrelevant). Finally,
89  // this square is colored red: (1,0,0,1) in RGBA.
90  m_batch.addGlyph(vec4(-25, 25, 50, 50), vec4(0), 0, vec4(1,0,0,1));
91  m_batch.end();
92 
93  // Actually draw stuff
94  m_simple_prog->use(); {
95  glUniformMatrix3fv(m_simple_prog->getUniformLocation("camera"), 1, GL_FALSE,
96  &m_camera.getCameraMatrix()[0][0]);
97  m_batch.render();
98  } m_simple_prog->unuse();
99 
100  m_window->swapBuffers();
101 }
