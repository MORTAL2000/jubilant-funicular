13 ILboolean ilRleCompressLine(ILubyte *p, ILuint n, ILubyte bpp,
14  ILubyte *q, ILuint *DestWidth, ILenum CompressMode) {
15 
16  ILint DiffCount; // pixel count until two identical
17  ILint SameCount; // number of identical adjacent pixels
18  ILint RLEBufSize = 0; // count of number of bytes encoded
19  ILint MaxRun;
20  const ILint bmp_pad_to_even = (ILint)(1 - ((size_t)q - *DestWidth) % 2);
21 
22  switch( CompressMode ) {
23  case IL_TGACOMP:
24  MaxRun = TGA_MAX_RUN;
25  break;
26  case IL_SGICOMP:
27  MaxRun = SGI_MAX_RUN;
28  break;
29  case IL_BMPCOMP:
30  MaxRun = BMP_MAX_RUN;
31  break;
32  default:
33  ilSetError(IL_INVALID_PARAM);
34  return IL_FALSE;
35  }
36 
37  while( n > 0 ) {
38  // Analyze pixels
39  DiffCount = CountDiffPixels(p, bpp, (ILint)n > MaxRun ? MaxRun : n);
40  SameCount = CountSamePixels(p, bpp, (ILint)n > MaxRun ? MaxRun : n);
41 
42  if( CompressMode == IL_BMPCOMP ) {
43  ILint remaining_data = n - DiffCount - SameCount;
44  if( remaining_data < 3 ) { // check if the run has gone near the end
45  // no absolute run can be done
46  // complete the line adding 0x01 + pixel, for each pixel
47  while( remaining_data > 0 ) {
48  *q++ = 0x01;
49  *q++ = *p++;
50  remaining_data--;
51  }
52  DiffCount = 0;
53  SameCount = 0;
54  n = 0;
55  }
56  }
57 
58  if( DiffCount > 0 ) { // create a raw packet (bmp absolute run)
59  switch(CompressMode) {
60  case IL_TGACOMP:
61  *q++ = (ILbyte)(DiffCount - 1);
62  break;
63  case IL_BMPCOMP:
64  *q++ = 0x00; RLEBufSize++;
65  *q++ = (ILbyte)DiffCount;
66  break;
67  case IL_SGICOMP:
68  *q++ = (ILbyte)(DiffCount | 0x80);
69  break;
70  }
71  n -= DiffCount;
72  RLEBufSize += (DiffCount * bpp) + 1;
73 
74  while( DiffCount > 0 ) {
75  switch(bpp) {
76  case 4: *q++ = *p++;
77  case 3: *q++ = *p++;
78  case 2: *q++ = *p++;
79  case 1: *q++ = *p++;
80  }
81  DiffCount--;
82  }
83 
84  if( CompressMode == IL_BMPCOMP ) {
85  if( (size_t)q % 2 == bmp_pad_to_even ) {
86  *q++ = 0x00; // insert padding
87  }
88  }
89  }
90 
91  if( SameCount > 1 ) { // create a RLE packet
92  switch(CompressMode) {
93  case IL_TGACOMP:
94  *q++ = (ILbyte)((SameCount - 1) | 0x80);
95  break;
96  case IL_SGICOMP:
97  case IL_BMPCOMP:
98  *q++ = (ILbyte)(SameCount);
99  break;
100  }
101  n -= SameCount;
102  RLEBufSize += bpp + 1;
103  p += (SameCount - 1) * bpp;
104  *q++ = *p++;
105  switch(bpp) {
106  case 4: *q++ = *p++;
107  case 3: *q++ = *p++;
108  case 2: *q++ = *p++;
109  case 1: *q++ = *p++;
110  }
111  }
112  }
113 
114  // write line termination code
115  switch(CompressMode) {
116  case IL_SGICOMP:
117  ++RLEBufSize;
118  *q++ = 0;
119  break;
120  case IL_BMPCOMP:
121  *q++ = 0x00; RLEBufSize++;
122  *q++ = 0x00; RLEBufSize++;
123  break;
124  }
125  *DestWidth = RLEBufSize;
126 
127  return IL_TRUE;
128 }

133  ILubyte *Dest, ILenum CompressMode, ILuint *ScanTable) {

135 
136  if( ScanTable )
137  imemclear(ScanTable,Depth*Height*sizeof(ILuint));
138  for( j = 0; j < Depth; j++ ) {
139  for( i = 0; i < Height; i++ ) {
140  if( ScanTable )
141  *ScanTable++ = DestW;
142  ilRleCompressLine(Data + j * SizeOfPlane + i * Bps, Width, Bpp, Dest + DestW, &LineLen, CompressMode);
143  DestW += LineLen;
144  }
145  }
146 
147  if( CompressMode == IL_BMPCOMP ) { // add end of image
148  *(Data+DestW) = 0x00; DestW++;
149  *(Data+DestW) = 0x01; DestW++;
150  }
151
