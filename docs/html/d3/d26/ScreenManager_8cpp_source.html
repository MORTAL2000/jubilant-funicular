141  Screen* currScreen = nullptr;
142  if (m_currScreen != -1) {
143  getCurrScreen()->onFocus(ScreenSwitchInfo(initFocusData));
144  }
145  while (m_currScreen != -1) {
146  currScreen = getCurrScreen();
147  // This while loop used to be neat
148  while (currScreen->getState() == ScreenState::RUNNING) {
149  m_limiter.begin();
150  CallbackManager::increment_frame();
151  update_input();
152  currScreen->update();
153  #ifdef NTA_USE_IMGUI
154  ImGui_ImplSdlGL3_NewFrame(m_window->getSDLWindow(GetSDLWindowKey()));
155  #endif
156  currScreen->render();
157  ErrorManager::handle_errors();
158  m_limiter.end();
159  }
160  switch(currScreen->getState()) {
161  case ScreenState::SWITCH: switchScreen(currScreen->getNextIndex()); break;
162  case ScreenState::SWITCH_ESC: switchScreen(currScreen->getEscIndex()); break;
163  case ScreenState::SWITCH_X: switchScreen(currScreen->getXIndex()); break;
164  case ScreenState::NONE: Logger::writeErrorToLog("state of screen " +
165  utils::to_string(m_currScreen) +
166  " is NONE",
167  IMPOSSIBLE_BEHAVIOR); break;
168  default: break; // should never happen
169  }
170  }
171  Logger::writeToLog("Exited ScreenManager");
172  }
